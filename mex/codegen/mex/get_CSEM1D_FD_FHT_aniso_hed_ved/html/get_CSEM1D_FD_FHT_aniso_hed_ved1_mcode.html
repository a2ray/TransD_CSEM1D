152            %for Hr
153            eyBase(irR,2*nF+1:3*nF)= (cst*[HrK(1:2:2*nF-1,iShift),HrK(2:2:2*nF,iShift)]*[Filter.J0;Filter.J1/rR(irR)])/rR(irR);
154            %for Eb
155            eyBase(irR,1*nF+1:2*nF)= (cst*[EbK(1:2:2*nF-1,iShift),EbK(2:2:2*nF,iShift)]*[Filter.J0;Filter.J1/rR(irR)])/rR(irR);
156            %for Er
157            eyBase(irR,1:nF)= ([cst*ErK(1:3:3*nF-2,iShift),cst*ErK(2:3:3*nF-1,iShift),sit*ErK(3:3:3*nF,iShift)]*[csb*Filter.J0;csb*Filter.J1/rR(irR);Filter.J1])/rR(irR);
158
159        end
160
161        % Compute spline coefficients:
162         %PP=spline(log10(rR) ,eyBase');
163         PPV=spline(log10(rR) ,eyBase',log10(rRx));
164         EH = 1i*zeros(6*nF,length(rRx));
165        % PPV = ppval(PP,log10(rRx));
166         EH = PPV;
167         
168         Er = EH(1:nF,:)';
169         Eb = sib*EH(  nF+1:2*nF,:)';
170         Hr = sib*EH(2*nF+1:3*nF,:)';
171         Hb = EH(3*nF+1:4*nF,:)';
172         Ez = EH(4*nF+1:5*nF,:)';
173         Hz = sib*EH(5*nF+1:6*nF,:)';
174 
175% and we're done
176end
